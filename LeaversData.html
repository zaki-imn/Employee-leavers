from seaborn import barplot
# Convert 'Date_of_Leaving' column to datetime format
series['Last Date of Employment'] = to_datetime(series['Last Date of Employment'])

# Extract year from the 'Date_of_Leaving' column
series['Year'] = series['Last Date of Employment'].dt.year
yearly_counts = series['Year'].value_counts().sort_index()

# Bar Chart: Number of Leavers per Year
figure(figsize=(10, 6))
barplot(x=yearly_counts.index, y=yearly_counts.values)
title('Number of Leavers per Year')
xlabel('Year')
